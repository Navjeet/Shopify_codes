<script>
     let arr = [];
  let arr2 = [];
  let t = $(".pagination__list li").length - 1;
  $(".pagination__list li").each(function(i){
    if(i > 0 && i < t){
      let n = parseInt($(this).text());
      arr.push(n);
    }
  });
  arr2 = arr.reverse();
  console.log(arr2);
    
  $(document).on('click','.u_load_btn_size',function(){
    let page = arr2.pop();
    let _url = "{{shop.url}}/collections/{{collection.handle}}?page=" + page;
    console.log(_url);
    $.ajax({
      url:_url,
      type: 'GET',
      dataType: 'html',
      beforeSend: function() {
        // setting a timeout
        $('.u_loader').removeClass('u_loader_hide');
        $('.u_loader_overlay').removeClass('u_loader_hide')
      },
      success: function(result){
        //           console.log('success',result)
        if(arr2.length <= 0){
          $(".u_load_btn_size").hide();
          $('.u_paginate_stats').hide();
        }
        let data = $(result).find(".u_featured_product_row").html();
        //           console.log(data,data);
        let stats =parseInt($('.u_paginate_items_show').text());
        let len =$(result).find(".u_fetured_product_block").length; 
        console.log(len);
        let paginate_items = stats + len;
        console.log('length',paginate_items);
        $(".u_featured_product_row").append(data);
        $(".u_paginate_items_show").text(paginate_items);

      },
      error : function(error) {
        console.log('error...',error)
      },
      complete: function() {
        $('.u_loader').addClass('u_loader_hide');
                $('.u_loader_overlay').addClass('u_loader_hide')
      }
    });
  });
</script>
