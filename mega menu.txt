  $(document).scroll(function(){
    var scroll =  $(this).scrollTop();
    // console.log(($(document).scrollTop() +'px'));
    if(scroll>0){
      $('.u_sticky').addClass('header_sticky');
    }
    else{
      $('.u_sticky').removeClass('header_sticky');
    }
  })
